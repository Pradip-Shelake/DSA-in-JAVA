package com.company;

public class B3 {
    public static void main(String[] args) {
        int[] letters = {'a', 'c', 'd' };
        int target = 'c';
        int ans = letters(letters, target);
        System.out.println(ans);

    }

    static int letters(int[] letters, int target) {
        int start = 0;
        int end = letters.length - 1;
        int mid = start + (end - start) / 2;
        while (start <= end) {
            if (target > letters[mid]) {
                return start = mid + 1;
            } else {
                return end = mid - 1;
            }


        }


        return letters[start % letters.length];
    }
}
